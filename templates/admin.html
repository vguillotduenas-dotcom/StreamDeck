@app.route('/admin', methods=['GET', 'POST']) @login_required def admin(): # 1. Sécurité : seul l'admin passe if current_user.role != 'admin': return redirect(url_for('index')) if request.method == 'POST': # Action : Créer un nouveau contenu (Film ou Série) if 'add_content' in request.form: v_type = request.form.get('type') v = Video( nom=request.form.get('nom'), img=request.form.get('img'), lien=request.form.get('lien_film'), type=v_type, genre=request.form.get('genre') ) db.session.add(v) db.session.flush() if v_type == 'serie' and request.form.get('saison'): ep = Episode( video_id=v.id, saison=request.form.get('saison'), titre_ep=request.form.get('titre_ep'), lien_ep=request.form.get('lien_ep') ) db.session.add(ep) # Action : Ajouter un épisode à une série existante elif 'add_only_ep' in request.form: ep = Episode( video_id=request.form.get('video_id'), saison=request.form.get('saison'), titre_ep=request.form.get('titre_ep'), lien_ep=request.form.get('lien_ep') ) db.session.add(ep) # Action : Générer un code invité elif 'gen_code' in request.form: db.session.add(User(code=str(uuid.uuid4())[:8].upper())) db.session.commit() # IMPORTANT : Après un POST, on redirige toujours pour "nettoyer" le formulaire return redirect(url_for('admin')) # 2. Le retour pour le mode "GET" (quand on arrive juste sur la page) # Vérifie bien que cette ligne est TOUT EN BAS et alignée avec le premier "if" films_liste = Video.query.all() users_liste = User.query.filter_by(role='user').all() return render_template('admin.html', films=films_liste, users=users_liste)
